<script>
    // Initialize the donation button
    BlinkPayButton.init({
        username: 'pretyflaco',
        containerId: 'blink-pay-button-container',
        themeMode: 'light',
        language: 'en',
        debug: true
    });
    
    // Check widget dimensions after initialization
    setTimeout(function() {
        const widget = document.querySelector('#blink-pay-button-container .blink-pay-widget');
        const button = document.querySelector('#blink-pay-button-container .blink-pay-button');
        const debugDiv = document.getElementById('debug-info');
        
        if (widget && button) {
            const widgetWidth = widget.offsetWidth;
            const buttonWidth = button.offsetWidth;
            const containerWidth = widget.parentElement.offsetWidth;
            const widgetCSSVar = getComputedStyle(widget).getPropertyValue('--blink-widget-width');
            
            const debugInfo = `Container: ${containerWidth}px | Widget: ${widgetWidth}px | Button: ${buttonWidth}px | CSS Var: ${widgetCSSVar}`;
            debugDiv.textContent = debugInfo;
        } else {
            debugDiv.textContent = 'Widget not found!';
        }
    }, 2000);
</script>
